write a program to print the palindrome primes in the given range 1 to input
            input : 150  output : 2 3 5 7 11 101 131

            var num = prompt("Enter a number to print palindrome primes : ");
            var result = "";
            for (let i = 2; i <= num; i++) {
                var count = 0;
                for (j = 2; j < i; j++) {
                    if (i % j == 0) {
                        count++;
                        break;
                    }
                }
                if (count == 0 && i == i.toString().split("").reverse().join(""))
                    result += i + " ";
            }
            console.log(`palindromic primes in the range of 1 to (${num}) is : ${result}`);
            document.write(`palindromic primes in the range of 1 to (${num}) is : ${result}`);
